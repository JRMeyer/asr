    \path (ss) 	edge		[right, above] 	node {$1$} 	(s1);

    \path (s9)  edge    	[loop above]    node {$i$}     (s9);

    \path (s1) 	edge		[right, above] 	    node {$1-a$} 	(s2);
    \path (s2) 	edge		[right, above]         node {$1-b$} 	(s3);
